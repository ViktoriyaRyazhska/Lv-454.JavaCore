package lesson;
import java.util.Scanner;
public class Person {
static Scanner sc = new Scanner(System.in);
private String fn, ln;
private int by, ty = 2019;

public Person(Scanner sc)
{
System.out.println("Print your fn");
this.fn = sc.nextLine();
System.out.println("Print your ln");
this.ln = sc.nextLine();
System.out.println("Print your by");
this.by = sc.nextInt();
}
public void getAge()
{
	System.out.println(ty - by);
}
public void input()
{
System.out.println("U're using your constructor for this.");	
}
public void output()
{
System.out.println(fn + ln + by);	
}
public void ChangeName()
{
	System.out.println("Print your fn");
	fn = sc.nextLine();
	System.out.println("Print your ln");
	ln = sc.nextLine();
	}

}
